  366   /**************************************************************************
  367    * FUNCTION :    ec_can_transmit                                           *
  368    *                                                                         *
  369    * DESCRIPTION : Copy CAN Tx_Buffer to CAN Message Buffer and then         *
  370    *               transmit message to CAN bus                               *
  371    *                                                                         *
  372    * PARAMETERS :  -Module: FLEXCAN_0, FLEXCAN_1 or FLEXCAN_2                *
  373    *               -BufNum: buffer number for transmit                       *
  374    *               -IdType: 0 means standard format;1 means entend format    *
  375    *               -Id: the message ID to be transmitted                     *
  376    *               -DataLength: data length of the frame data                *
  377    *               -Tx_Buffer: the data array to be transmitted              *
  378    *                                                                         *
  379    * RETURN :      -result: CAN_BUSY or CAN_OK.                              *
  380    ***************************************************************************/
  381   CAN_Return ec_can_transmit(uint8 Module,uint8 BufNum,uint8 IdType,uint32 Id,
  382     uint8 DataLength,uint8 Tx_Buffer[])
  383   {
  384     uint8 i;
  385     CAN_Return result = CAN_BUSY;
  386     if (BufNum<8)
  387       BufNum = 8;
  388     else if ((BufNum>15)&(BufNum<56))
  389       BufNum = 15;
  390     else if (BufNum>63)
  391       BufNum = 63;
  392     else {
  393       if (FLEXCAN(Module).MB[BufNum].CS.B.CODE == TX_MB_CODE_TRANSMIT) {
  394         result = CAN_BUSY;
  395       } else {
  396         if (IdType == 0) {
  397           FLEXCAN(Module).MB[BufNum].CS.B.IDE = MB_IDType_ST;
  398           FLEXCAN(Module).MB[BufNum].ID.B.ID_STD = Id;
  399           FLEXCAN(Module).MB[BufNum].CS.B.SRR = 0;
  400         } else {
  401           FLEXCAN(Module).MB[BufNum].CS.B.IDE = MB_IDType_EX;
  402           FLEXCAN(Module).MB[BufNum].ID.R = (uint32)(Id&0x1FFFFFFF);
  403           FLEXCAN(Module).MB[BufNum].CS.B.SRR = 1;
  404         }
  405   
  406         FLEXCAN(Module).MB[BufNum].CS.B.RTR = 0;
  407         for (i=0; i<DataLength; i++) {
  408           FLEXCAN(Module).MB[BufNum].DATA.B[i] = Tx_Buffer[i];
  409         }
  410   
  411         FLEXCAN(Module).MB[BufNum].CS.B.DLC = DataLength;
  412         FLEXCAN(Module).MB[BufNum].CS.B.CODE = TX_MB_CODE_TRANSMIT;
  413         result = CAN_OK;
  414         ec_can_appltxconfirmation(Module,BufNum);
  415       }
  416     }
  417   
  418     return(result);
  419   }
